from streamlit_lottie import st_lottie
import requests

def load_lottieurl(url):
    r = requests.get(url)
    if r.status_code != 200:
        return None
    return r.json()

animation = load_lottieurl("https://assets10.lottiefiles.com/packages/lf20_jzv1z8cu.json")

if animation:
    st_lottie(animation, height=300)
